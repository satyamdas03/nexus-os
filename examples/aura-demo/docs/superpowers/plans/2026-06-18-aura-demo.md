# AURA Demo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy AURA — an AI layer on top of a portfolio-assurance platform — that takes a client portfolio RED → UNDERSTOOD → REMEDIATED → GREEN with a deterministic rules engine as source of truth, Claude agents grounded against it, human-in-the-loop approval, append-only audit log, and a Hermes-style learning loop.

**Architecture:** Next.js + FastAPI monorepo. A pure `rules_engine.py` is the only source of truth for compliance (colours blocks, verifies remediations). Two Claude agents (Explainer, Remediation) are grounded against the engine's `RulesResult` so they cannot hallucinate compliance. Remediation runs a closed verify loop (propose → simulate → re-check → one retry). Every action appends to `audit.jsonl`. A `reflect.py` loop surfaces human-approved preference patterns. Frontend renders the engine output alongside AI output with explicit deterministic-vs-advisory tags.

**Tech Stack:** Next.js (App Router) + TypeScript + Tailwind + recharts (frontend); FastAPI + Python (backend); Anthropic Claude Sonnet (LLM, env `ANTHROPIC_API_KEY`); JSON files + JSONL audit log (no external DB); Render (backend) + Vercel (frontend).

**Spec:** `docs/superpowers/specs/2026-06-18-aura-demo-design.md` — read it first.

## Global Constraints

- Synthetic/public data only. Never real Financial Simplicity data.
- Deterministic rules engine is the source of truth. LLM never decides compliance.
- Human-in-the-loop: no action auto-executes. Every consequential step has explicit Approve/Modify.
- Every AI claim grounded against `RulesResult`; every decision logged to `audit.jsonl` with timestamp + actor + rationale.
- Trust-first aesthetic: calm, muted, finance-grade, generous whitespace.
- AURA is a layer on top, not a rebuild — frame accordingly in copy.
- Python tests via `pytest`; frontend tests via `vitest` + `@testing-library/react`.
- Frequent commits, one logical change per commit, conventional commit messages.
- `RulesResult` contract (from spec §3) is the stable interface every task builds against:

```python
RulesResult = {
  status: "green" | "orange" | "red",
  breaches: [Breach],      # red-level
  watches: [Breach],       # orange-level drift
  per_rule: [{rule, pass, current, limit, offending_holdings, severity}]
}
Breach = {rule, current, limit, offending_holdings, severity, plain}
```

**Parallelization note:** Tasks 1–8 are sequential (contract + backend). Tasks 9–12 (frontend screens) are parallelizable after Task 8. Tasks 13–14 (Hermes loop) parallelizable after Task 8. Task 15–16 last.

`backend/core/rules_engine.py`:
```python
"""Deterministic source of truth. LLM never overrides this."""
from typing import Any

def _total_value(portfolio: dict) -> float:
    return sum(h["market_value"] for h in portfolio["holdings"]) + portfolio["cash"]

def _weights(portfolio: dict) -> dict[str, float]:
    total = _total_value(portfolio)
    if total == 0:
        return {}
    w: dict[str, float] = {}
    for h in portfolio["holdings"]:
        w[h["ticker"]] = h["market_value"] / total
    return w

def _asset_class_weights(portfolio: dict) -> dict[str, float]:
    total = _total_value(portfolio)
    if total == 0:
        return {}
    w: dict[str, float] = {}
    for h in portfolio["holdings"]:
        w[h["asset_class"]] = w.get(h["asset_class"], 0.0) + h["market_value"] / total
    return w

def _sector_weights(portfolio: dict) -> dict[str, float]:
    total = _total_value(portfolio)
    if total == 0:
        return {}
    w: dict[str, float] = {}
    for h in portfolio["holdings"]:
        w[h["sector"]] = w.get(h["sector"], 0.0) + h["market_value"] / total
    return w

def _cash_ratio(portfolio: dict) -> float:
    total = _total_value(portfolio)
    return portfolio["cash"] / total if total else 0.0

def check(portfolio: dict, mandate: dict) -> dict:
    """Pure function. Returns RulesResult."""
    per_rule: list[dict] = []
    breaches: list[dict] = []
    watches: list[dict] = []

    acw = _asset_class_weights(portfolio)
    for ac, limit in mandate.get("max_asset_class_weight", {}).items():
        current = acw.get(ac, 0.0)
        offending = [h["ticker"] for h in portfolio["holdings"] if h["asset_class"] == ac and current > limit + 1e-9]
        passed = current <= limit + 1e-9
        sev = "red" if not passed else "green"
        per_rule.append({"rule": f"max_asset_class_weight:{ac}", "pass": passed, "current": current, "limit": limit, "offending_holdings": offending, "severity": sev})
        if not passed:
            breaches.append({"rule": f"max_asset_class_weight:{ac}", "current": current, "limit": limit, "offending_holdings": offending, "severity": "red", "plain": f"{ac} {current*100:.0f}% > {limit*100:.0f}% cap"})

    sw = _sector_weights(portfolio)
    for sec, limit in mandate.get("max_sector_weight", {}).items():
        current = sw.get(sec, 0.0)
        offending = [h["ticker"] for h in portfolio["holdings"] if h["sector"] == sec and current > limit + 1e-9]
        passed = current <= limit + 1e-9
        sev = "red" if not passed else "green"
        per_rule.append({"rule": f"max_sector_weight:{sec}", "pass": passed, "current": current, "limit": limit, "offending_holdings": offending, "severity": sev})
        if not passed:
            breaches.append({"rule": f"max_sector_weight:{sec}", "current": current, "limit": limit, "offending_holdings": offending, "severity": "red", "plain": f"{sec} {current*100:.0f}% > {limit*100:.0f}% cap"})

    approved = set(mandate.get("approved_universe", []))
    offending_univ = [h["ticker"] for h in portfolio["holdings"] if h["ticker"] not in approved]
    passed_univ = len(offending_univ) == 0
    per_rule.append({"rule": "approved_universe", "pass": passed_univ, "current": offending_univ, "limit": list(approved), "offending_holdings": offending_univ, "severity": "red" if not passed_univ else "green"})
    if not passed_univ:
        breaches.append({"rule": "approved_universe", "current": offending_univ, "limit": list(approved), "offending_holdings": offending_univ, "severity": "red", "plain": f"{', '.join(offending_univ)} not in approved list"})

    weights = _weights(portfolio)
    max_single = mandate.get("max_single_holding", 1.0)
    offending_single = [t for t, w in weights.items() if w > max_single + 1e-9]
    passed_single = len(offending_single) == 0
    per_rule.append({"rule": "max_single_holding", "pass": passed_single, "current": max(weights.values()) if weights else 0.0, "limit": max_single, "offending_holdings": offending_single, "severity": "red" if not passed_single else "green"})
    if not passed_single:
        breaches.append({"rule": "max_single_holding", "current": max(weights.values()), "limit": max_single, "offending_holdings": offending_single, "severity": "red", "plain": f"Single holding {max(weights.values())*100:.0f}% > {max_single*100:.0f}% cap"})

    cash_ratio = _cash_ratio(portfolio)
    min_cash = mandate.get("min_cash", 0.0)
    passed_cash = cash_ratio >= min_cash - 1e-9
    per_rule.append({"rule": "min_cash", "pass": passed_cash, "current": cash_ratio, "limit": min_cash, "offending_holdings": [], "severity": "red" if not passed_cash else "green"})
    if not passed_cash:
        breaches.append({"rule": "min_cash", "current": cash_ratio, "limit": min_cash, "offending_holdings": [], "severity": "red", "plain": f"Cash {cash_ratio*100:.1f}% < {min_cash*100:.1f}% min"})

    target = mandate.get("target_allocation", {})
    tol = mandate.get("drift_tolerance", 0.05)
    for ac, tgt in target.items():
        current = acw.get(ac, 0.0)
        drift = abs(current - tgt)
        if drift > tol + 1e-9:
            offending = [h["ticker"] for h in portfolio["holdings"] if h["asset_class"] == ac]
            per_rule.append({"rule": f"drift:{ac}", "pass": False, "current": current, "limit": tgt, "offending_holdings": offending, "severity": "orange"})
            watches.append({"rule": f"drift:{ac}", "current": current, "limit": tgt, "offending_holdings": offending, "severity": "orange", "plain": f"{ac} drifted {drift*100:.0f}% from {tgt*100:.0f}% target"})

    status = status_of({"breaches": breaches, "watches": watches, "per_rule": per_rule})
    return {"status": status, "breaches": breaches, "watches": watches, "per_rule": per_rule}

def status_of(rules_result: dict) -> str:
    if rules_result.get("breaches"):
        return "red"
    if rules_result.get("watches"):
        return "orange"
    return "green"
```

"""LLM provider interface. MockLLM for tests/offline; ClaudeProvider for real calls.
The agents never call the LLM directly to decide compliance — they're grounded
against rules_engine output. The LLM only drafts prose/proposals."""
import os
from typing import Protocol

class LLMProvider(Protocol):
    def complete(self, system: str, user: str) -> str: ...

class MockLLM:
    """Deterministic. Returns a structured echo so tests can assert grounding."""
    def complete(self, system: str, user: str) -> str:
        return f"[MOCK LLM] grounded explanation for: {user[:120]}"

class ClaudeProvider:
    def __init__(self, api_key: str | None = None, model: str = "claude-sonnet-4-6"):
        self._api_key = api_key or os.environ.get("ANTHROPIC_API_KEY", "")
        self._model = model
        self._client = None

    def complete(self, system: str, user: str) -> str:
        if self._client is None:
            from anthropic import Anthropic
            self._client = Anthropic(api_key=self._api_key)
        msg = self._client.messages.create(
            model=self._model, max_tokens=1024, system=system, messages=[{"role":"user","content":user}],
        )
        return "".join(b.text for b in msg.content if hasattr(b, "text"))

def get_llm() -> LLMProvider:
    if os.environ.get("ANTHROPIC_API_KEY"):
        return ClaudeProvider()
    return MockLLM()
```

"""Explainer agent. Grounded against rules_engine output — may only describe
breaches/watches the engine flagged. Never invents rules or numbers."""
import json
from typing import Any
from agents.llm import LLMProvider, get_llm

SYSTEM = ("You are an assurance explainer for a wealth-management platform. "
          "You are given a portfolio, its mandate, and the DETERMINISTIC rules-engine result. "
          "You may ONLY describe the breaches and watches the engine flagged. "
          "Do not invent rules, numbers, or breaches. Plain English, calm, 2-4 sentences. "
          "Start with the count of breaches and watches. Never claim a rule is broken if the engine did not flag it.")

def _facts(portfolio: dict, rules_result: dict) -> str:
    return json.dumps({
        "client": portfolio["client_name"],
        "fum": portfolio["fum"],
        "holdings": [{"ticker": h["ticker"], "weight_pct": round(h["market_value"]/(sum(x["market_value"] for x in portfolio["holdings"])+portfolio["cash"])*100,1)} for h in portfolio["holdings"]],
        "cash_pct": round(portfolio["cash"]/(sum(x["market_value"] for x in portfolio["holdings"])+portfolio["cash"])*100,1),
        "engine_breaches": rules_result["breaches"],
        "engine_watches": rules_result["watches"],
    }, ensure_ascii=False)

def explain(portfolio: dict, rules_result: dict, llm: LLMProvider | None = None) -> dict:
    llm = llm or get_llm()
    facts = _facts(portfolio, rules_result)
    user = f"Rules-engine result (GROUND TRUTH — do not contradict):\n{facts}\n\nWrite the plain-English assurance narrative."
    narrative = llm.complete(SYSTEM, user)
    summaries = [b["plain"] for b in rules_result["breaches"]] + [w["plain"] for w in rules_result["watches"]]
    return {"narrative": narrative, "breach_summaries": summaries}
```

```python
"""Remediation agent + closed verify loop. AI proposes, rules_engine disposes.
The resulting portfolio is re-checked by the deterministic engine; if still
red/orange, the agent gets ONE retry. The engine — not the LLM — is final."""
import json
from agents.llm import LLMProvider, get_llm
from core.rules_engine import check

SYSTEM = ("You are a portfolio remediation agent. Given a portfolio, its mandate, and the "
          "deterministic rules-engine result, propose the MINIMAL set of compliant trades to "
          "bring the portfolio to GREEN. Trades must use tickers from the approved universe and "
          "respect max_single_holding and min_cash. Output STRICT JSON only: a list of objects "
          "with keys ticker, action ('buy'|'sell'), units (number), value (number), rationale (str). "
          "Do not include prose outside the JSON. Never sell tickers not held; never buy unapproved tickers.")

def _facts(portfolio: dict, rules_result: dict) -> str:
    return json.dumps({"holdings": portfolio["holdings"], "cash": portfolio["cash"],
                       "mandate": portfolio["mandate"], "engine_breaches": rules_result["breaches"]},
                      ensure_ascii=False)

def _apply_trades(portfolio: dict, trades: list[dict]) -> dict:
    import copy
    p = copy.deepcopy(portfolio)
    by_ticker = {h["ticker"]: h for h in p["holdings"]}
    for t in trades:
        tk = t.get("ticker"); act = t.get("action"); units = float(t.get("units", 0))
        price = next((h["price"] for h in p["holdings"] if h["ticker"]==tk), None)
        if price is None and act == "buy":
            # unknown price: skip (guard); real provider returns approved tickers only
            continue
        if act == "sell" and tk in by_ticker:
            h = by_ticker[tk]
            h["units"] = max(0, h["units"] - units)
            h["market_value"] = round(h["units"] * h["price"], 2)
            p["cash"] = round(p["cash"] + units * h["price"], 2)
            if h["units"] <= 1e-6:
                p["holdings"] = [x for x in p["holdings"] if x["ticker"] != tk]
        elif act == "buy":
            mv = round(units * price, 2)
            if tk in by_ticker:
                by_ticker[tk]["units"] += units
                by_ticker[tk]["market_value"] = round(by_ticker[tk]["units"] * price, 2)
            else:
                # add a minimal holding using approved universe metadata
                uni = {u["ticker"]: u for u in _UNIVERSE_LOOKUP}
                meta = uni.get(tk, {"name":tk,"asset_class":"Equity","sector":"Broad"})
                p["holdings"].append({"ticker":tk,"name":meta["name"],"asset_class":meta["asset_class"],
                                      "sector":meta["sector"],"units":units,"price":price,"market_value":mv})
            p["cash"] = max(0, round(p["cash"] - mv, 2))
    return p

# minimal price/sector lookup for buys (must match generate_data universe)
_UNIVERSE_LOOKUP = [
    {"ticker":"SPY","name":"S&P 500 ETF","asset_class":"Equity","sector":"Broad","price":500.0},
    {"ticker":"QQQ","name":"Nasdaq 100 ETF","asset_class":"Equity","sector":"Broad","price":420.0},
    {"ticker":"XLV","name":"Healthcare ETF","asset_class":"Equity","sector":"Healthcare","price":145.0},
    {"ticker":"XLF","name":"Financials ETF","asset_class":"Equity","sector":"Financials","price":45.0},
    {"ticker":"TLT","name":"20+ Yr Treasury","asset_class":"Bonds","sector":"Govt","price":95.0},
    {"ticker":"GLD","name":"Gold ETF","asset_class":"Commodity","sector":"Metals","price":240.0},
]

def _parse_trades(text: str) -> list[dict]:
    # MockLLM returns [MOCK LLM]... — handle gracefully as empty trades for tests
    try:
        start = text.index("["); end = text.rindex("]") + 1
        return json.loads(text[start:end])
    except (ValueError, json.JSONDecodeError):
        return []

def remediate(portfolio: dict, rules_result: dict, llm: LLMProvider | None = None) -> dict:
    llm = llm or get_llm()
    facts = _facts(portfolio, rules_result)
    trades = _parse_trades(llm.complete(SYSTEM, facts))
    resulting = _apply_trades(portfolio, trades)
    verification = check(resulting, portfolio["mandate"])
    retried = False
    if verification["status"] != "green" and verification["breaches"]:
        # one retry, feeding back what's still wrong
        retry_prompt = facts + "\n\nPrevious proposal still leaves these breaches: " + json.dumps(verification["breaches"]) + "\nRevise the trades."
        trades = _parse_trades(llm.complete(SYSTEM, retry_prompt))
        resulting = _apply_trades(portfolio, trades)
        verification = check(resulting, portfolio["mandate"])
        retried = True
    return {"trades": trades, "verification": verification,
            "resulting_portfolio": resulting, "retried": retried,
            "resolved": verification["status"] == "green"}
```

```python
"""Hermes-style learning loop. Surfaces human-approved preference patterns.
Suggests ONE rule-of-thumb at a time. Never auto-applies — human must Adopt."""
import json
from pathlib import Path
from collections import Counter

_AUDIT = Path(__file__).parent.parent / "data" / "audit.jsonl"
_PREFS = Path(__file__).parent.parent / "data" / "preferences.jsonl"
_THRESHOLD = 5  # N decisions of a breach type before suggesting

def _seed_audit(entries: list[dict]) -> None:
    """Test helper — overwrite audit log with given entries."""
    _AUDIT.parent.mkdir(parents=True, exist_ok=True)
    _AUDIT.write_text("\n".join(json.dumps(e) for e in entries) + ("\n" if entries else ""))

def _history() -> list[dict]:
    if not _AUDIT.exists(): return []
    return [json.loads(l) for l in _AUDIT.read_text().splitlines() if l.strip()]

def suggest(client_id: str) -> dict | None:
    """Detect a preference pattern from past approve decisions. Returns suggestion or None."""
    approves = [e for e in _history() if e.get("action_type") == "approve"]
    by_breach: dict[str, list[str]] = {}
    for e in approves:
        bt = e.get("payload", {}).get("breach_type")
        choice = e.get("payload", {}).get("choice")
        if bt and choice:
            by_breach.setdefault(bt, []).append(choice)
    for bt, choices in by_breach.items():
        if len(choices) >= _THRESHOLD:
            top, n = Counter(choices).most_common(1)[0]
            if n >= _THRESHOLD:
                return {"breach_type": bt, "pattern": f"{n} of {len(choices)} {bt} cases: {top}",
                        "suggestion": f"In {n} of last {len(choices)} {bt.split(':')[-1]} cases you {top} — default to that?",
                        "count": n}
    return None

def adopt(preference: dict) -> None:
    """Human clicked Adopt — write preference + rationale, version bump."""
    _PREFS.parent.mkdir(parents=True, exist_ok=True)
    with _PREFS.open("a") as f:
        f.write(json.dumps(preference) + "\n")
```
